Installation

# Extract the archive
tar -xzf kubectl-oadp_*.tar.gz
# Make it executable
chmod +x kubectl-oadp
# Move to your PATH
sudo mv kubectl-oadp /usr/local/bin/
# Verify it works
oc oadp --help
